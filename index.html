Perdiste"); updateMoney(); return; } tLevel++; multiplier++; document.getElementById("level").innerText = tLevel; document.getElementById("mult").innerText = multiplier; msg("✅ Safe"); } function cashout(){ if(!playingTowers) return; let bet = getBet(); let win = bet * multiplier; money += win; playingTowers = false; msg("💰 Ganaste " + win); updateMoney(); } // ---------------- LEADERBOARD ---------------- function saveScore(){ let name = document.getElementById("playerName").value || "Anon"; let scores = JSON.parse(localStorage.getItem("scores")) || []; scores.push({name: name, money: money}); scores.sort((a,b)=>b.money-a.money); scores = scores.slice(0,5); // top 5 localStorage.setItem("scores", JSON.stringify(scores)); showScores(); } function showScores(){ let scores = JSON.parse(localStorage.getItem("scores")) || []; let list = document.getElementById("scores"); list.innerHTML = ""; scores.forEach(s=>{ let li = document.createElement("li"); li.innerText = s.name + " - 💰 " + s.money; list.appendChild(li); }); } // cargar ranking al entrar showScores(); updateMoney();
